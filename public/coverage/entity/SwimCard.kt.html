package entity

data class SwimCard (val suit: CardSuit, val value: CardValue) {
    override fun toString() = "$suit$value"

    public fun valueOf():Float{
        when(this.value){
            CardValue.TWO -> return 2F;
            CardValue.THREE -> return 3F;
            CardValue.FOUR -> return 4F;
            CardValue.FIVE -> return 5F;
            CardValue.SIX -> return 6F;
            CardValue.SEVEN -> return 7F;
            CardValue.EIGHT -> return 8F;
            CardValue.NINE -> return 9F;
            CardValue.TEN -> return 10F;
            CardValue.JACK -> return 10F;
            CardValue.QUEEN -> return 10F;
            CardValue.KING -> return 10F;
            CardValue.ACE -> return 11F;
        }
    }
}
