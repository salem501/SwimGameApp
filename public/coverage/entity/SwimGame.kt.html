package entity

/**
 * diese Klasse ist eine Entität, die das ganze Spiel repräsentiert.
 * @param poolCards die Karten, die im Mittel liegen. Sie müssen 3 sein
 * @param drawStack die Decke, von der man ziehen kann
 * @param players die Spieler, sie müssen zwischen 2 und 4 sein
 */
data class SwimGame (val poolCards:List<SwimCard>,
                val drawStack:List<SwimCard>,
                val players:List<SwimPlayer>
                ){
    init {
        if((poolCards.size != 3)|| (players.size<2)||(players.size>4) ){
            throw java.lang.IllegalArgumentException("Ungültige Eingaben")
        }
    }
}
